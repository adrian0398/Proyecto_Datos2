382  GstState old_state, new_state, pending_state;
383  gst_message_parse_state_changed (msg, &old_state, &new_state, &pending_state);
384  if (GST_MESSAGE_SRC (msg) == GST_OBJECT (data->playbin)) {
385  data->state = new_state;
386  g_print ("State set to %s\n", gst_element_state_get_name (new_state));
387  if (old_state == GST_STATE_READY && new_state == GST_STATE_PAUSED) {
